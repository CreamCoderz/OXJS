  6 /**
  7  * Base object for OXJS. All other objects inherit from this one.
  8  * @namespace
  9  */
 10 OX.Base = {
 11   /**
 12    * Creates a new object which extends the current object.  Any
 13    * arguments are mixed in to the new object as if +OX.Base.extend+ was
 14    * called on the new object with remaining args.
 15    *
 16    * @returns {OX.Base} the new object
 17    */
 18   extend: function () {
 19     var F = function () {};
 20     F.prototype = this;
 21 
 22     var rc = new F();
 23     rc.mixin.apply(rc, arguments);
 24 
 25     return rc;
 26   },
 27 
 28   /**
 29    * Iterates over all arguments, adding their own properties to the
 30    * receiver.
 31    *
 32    * @returns {OX.Base} the receiver
 33    */
 34   mixin: function () {
 35     for (var i = 0, len = arguments.length; i < len; i++) {
 36       for (var k in arguments[i]) if (arguments[i].hasOwnProperty(k)) {
 37         this[k] = arguments[i][k];
 38       }
 39     }
 40 
 41     return this;
 42   }
 43 };

 90     return OX.URI.Base.extend();

 92 
 93   /**
 94    * Convert +object+ into an OX.URI.Base object

 97    * @returns {OX.URI.Base} A new URI object
 98    */
